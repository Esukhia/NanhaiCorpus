རྒྱ་སེ་མེ་ཏོག དང ཕྱེ་མ་ལེབ ། ༄༅ ། ། གནའ་སྔ་མོ ཕྱེ་མ་ལེབ རྐྱང་ཕྱག འཚལིག རྒྱ་སེ་མེ་ཏོག ལ ཧ་རྐྱང་ཕྱག འཚལང གི དགའ་པོ ཡོད ། རྒྱ་སེ་མེ་ཏོག ཀྱང ཕྱེ་མ་ལེབ ལ ཧ་རྐྱང་ཕྱག འཚལང དགའ་པོ བྱུང ། ཕྱེ་མ་ལེབ ལ མཛེས་སྡུག ལྡན་པ འི གཤོག་པ གཉིས ཡོད ། རྒྱ་སེ་མེ ཏོག ལ མངར་བསིལ ལྡན་པ འི དྲི་ཞིམ དེ ཡོད་པ དེ ར ཕྱེ་མ་ལེབ ཀྱི ཡིད་ཀྱི་དབང་པོ རྐྱང་ཕྱག འཚལིག་རྐྱང་ཕྱག འཚལར ཏུ འཕྲོག གི ཡོད ། ཕྱེ་མ་ལེབ དེ རྒྱ་སེ་མེ་ཏོག གི འགྲམ དུ སླེབས མ་ཐག ང་རང ཁྱེད ལ ཧ་རྐྱང་ཕྱག འཚལང དགའ་པོ ཡོད ། རྐྱང་ཕྱག འཚལེས བཤད་པ ར རྒྱ་སེ་མེ་ཏོག གིས དགའ་སྤྲོ འི ངང མགོ་བོ ལྕོག་ལྕོག བྱེད ཀྱི ཡོད ། ཁོང་གཉིས ག་དུས ཡིན་ན འི མཉམ་དུ འཛོམས་པ དང ལྷད་མེད ཀྱི བརྩེ་སེམས ལ འགྱུར་བ མེད རྐྱང་ཕྱག འཚལེས ཤོད ཀྱི ཡོད ཉིན་ཞིག ཕྱེ་མ་ལེབ རྒྱ་སེ་མེ་ཏོག རྩ ར ཡོང་དུས གློ་བུར དུ ཁོང་ཁྲོ ཆེན་པོ ལངས ནས འགྲོ་མཚམས བཞག རྒྱ་སེ་མེ་ཏོག ནི ཕྱེ་མ་ལེབ ཐུག འདོད ཀྱིས ཡུན་རིང བསྒུགས ནས བསྡད ། འོན་ཀྱང ཕྱེ་མ་ལེབ དེ ཡུན་རིང་པོ ཡོང་མཁན་མེད ། ཟླ་བ གརྐྱང་ཕྱག འཚལིག གི རྗེས་ལ ཕྱེ་མ་ལེབ དེ གཞི་ནས སླེབས་པ རེད ། རྒྱ་སེ་མེ་ཏོག གིས བཤད་དོན ། ཁྱེད ཀྱིས ང ལ ལྷད་མེད ཀྱི བརྩེ དུང ཞིག བརྐྱང་ཕྱག འཚལངས ཡོད་དམ ། ང ས ཁྱེད་རང མེ་ཏོག གཞན དང མཉམ་དུ རྩེད་མོ རྩེས་པ དང ཟེའུ་འབྲུ སྦྱོར་སྡེབ བྱེད་པ མཐོང བྱུང ཞེས བཤད ། ཕྱེ་མ་ལེབ ཀྱིས ། ཁྱེད ཀྱིས ང ལ སྐད་ཆ དེ འདྲ ཤོད ཀྱི ཡོད་དམ ། ཁྱེད་རང ལ སྦྲང་ནོར་བུ ས བསྐོར་བ བརྒྱབ་པ དང དགའ་དགའ་སྤྲོ་སྤྲོ འི ངང གྲོགས་པོ བསྒྲིག་པ ད་དུང ཟེའུ་འབྲུ ཁོང་ཚོ ར འཁྱེར བརྐྱང་ཕྱག འཚལུག་པ ང ས མཐོང བྱུང ། ཞེས བཤད ། ནང་དོན ། རང་གི སྐྱོན ལ སྔོན་ལ ལྟོས །
